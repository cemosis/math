  \only<1>{Équations de Maxwell
  $$
  \begin{aligned}
    \frac{\partial B}{\partial t} + \nabla \times E = 0\\
    -\frac{\partial D}{\partial t} + \nabla \times H = J\\
    \nabla \cdot B = 0, \nabla \cdot D = \rho\\
    B = \mu H, D = \epsilon E, J = \sigma E
  \end{aligned}
  $$}
